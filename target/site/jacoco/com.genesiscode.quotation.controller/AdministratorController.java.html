package com.genesiscode.quotation.controller;
import com.genesiscode.quotation.domain.unit.UnitAdministrative;
import com.genesiscode.quotation.service.AdministratorService;
import org.springframework.web.bind.annotation.*;
import java.util.*;
import lombok.*;

@AllArgsConstructor
@RestController @RequestMapping(path = "/api/administrator")
public class AdministratorController {

    private final AdministratorService administratorService;

    @PostMapping
    public void createUnitAdministrative(UnitAdministrative unit) {
        administratorService.createUnitAdministrative(unit);
    }

    @GetMapping
    public List<UnitAdministrative> getUnitsAdministrative() {
        return administratorService.getUnitsAdministrative();
    }

    @GetMapping(path = "{id}")
    public UnitAdministrative getUnitAdministrativeById(@PathVariable("id") Long id) {
        return administratorService.getUnitAdministrativeById(id);
    }
}
